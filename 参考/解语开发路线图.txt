濒危语言协作平台：核心数据库设计 (DLx 兼容版)
SQL
-- 1. 语料项目容器 (对应 DLx Text 概念)
CREATE TABLE texts (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    title JSONB NOT NULL, -- 存储多语言标题: {"eng": "Story A", "zho": "故事 A"}
    metadata JSONB,       -- 存储 DLx 规范的元数据
    language_code VARCHAR(10), -- ISO 639-3 代码
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 2. 媒体资源表 (实现音视频联动的基础)
CREATE TABLE media_items (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    text_id UUID REFERENCES texts(id) ON DELETE CASCADE,
    filename VARCHAR(255) NOT NULL,
    url TEXT,            -- 远程或本地路径
    details JSONB,       -- 采样率、编码格式、总时长等
    is_offline_cached BOOLEAN DEFAULT FALSE
);

-- 3. 句子/话语表 (对应 DLx Utterance 概念)
CREATE TABLE utterances (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    text_id UUID REFERENCES texts(id) ON DELETE CASCADE,
    media_id UUID REFERENCES media_items(id),
    
    -- DLx 核心对齐字段 (SS.MMM 格式在程序中处理，存储为浮点数提高检索效率)
    start_time NUMERIC(10, 3) NOT NULL, 
    end_time NUMERIC(10, 3) NOT NULL,
    
    -- 语言学内容
    transcription JSONB, -- {"IPA": "...", "Mod": "..."}
    translation JSONB,   -- {"eng": "...", "zho": "..."}
    
    -- AI 决策元数据
    ai_confidence FLOAT CHECK (ai_confidence BETWEEN 0 AND 1),
    ai_mode VARCHAR(20), -- 'AUTO', 'SUGGEST'
    ai_reasoning TEXT,
    
    is_verified BOOLEAN DEFAULT FALSE -- 人工是否审核通过
);

-- 4. 词典词条表 (对应 DLx Lexeme 概念)
CREATE TABLE lexemes (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    lemma JSONB NOT NULL,      -- 核心词头: {"IPA": "kušt"}
    lexeme_type VARCHAR(50),   -- lexical, grammatical
    morpheme_type VARCHAR(50), -- stem, prefix, suffix
    
    -- 深度嵌套数据使用 JSONB 存储，保持 DLx 完整性
    forms JSONB DEFAULT '[]'::jsonb,   -- 存储异体形式 (Forms) 与变体 (Allomorphs)
    senses JSONB DEFAULT '[]'::jsonb,  -- 存储语义 (Senses) 与标注 (Gloss)
    
    -- AI 与统计
    ai_metadata JSONB,
    usage_count INTEGER DEFAULT 0
);

-- 5. 语料与词典交叉引用 (实现“检索句子->点击词语->查看词典”联动)
CREATE TABLE corpus_lexicon_links (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    utterance_id UUID REFERENCES utterances(id),
    lexeme_id UUID REFERENCES lexemes(id),
    word_index INTEGER, -- 词在句子中的位置
    confidence FLOAT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 6. 非语言学标注 (对应 DLx Annotation 概念)
CREATE TABLE annotations (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    text_id UUID REFERENCES texts(id),
    annotation_type VARCHAR(20), -- 'timespan', 'timestamp'
    start_time NUMERIC(10, 3),
    end_time NUMERIC(10, 3),
    ts NUMERIC(10, 3),          -- 针对 timestamp 类型
    tags JSONB,                 -- {"creakyVoice": true}
    notes JSONB                 -- 记录观察笔记
);

-- 创建性能索引
CREATE INDEX idx_utterance_time ON utterances (media_id, start_time, end_time);
CREATE INDEX idx_lexemes_lemma ON lexemes USING gin (lemma);
CREATE INDEX idx_utterance_transcription ON utterances USING gin (transcription);

/**
 * 核心功能：处理用户新录入的转写内容
 * 流程：分词 -> 词库匹配 -> LLM 推理 -> 置信度评分 -> 自动确认/留待审核
 */
async function processTranscription(utteranceId, rawText) {
    // 1. 从数据库获取该句子的上下文信息（前后文、多媒体元数据）
    const utterance = await db.utterances.get(utteranceId);
    
    // 2. 调度 AI 引擎（根据用户设置选择 Cloud 或 Local）
    const aiEngine = userConfig.aiMode === 'cloud' ? cloudLLM : localLLM;
    
    // 3. 执行 AI 分析：请求模型进行隔行标注建议与一致性检查
    const aiResult = await aiEngine.analyze({
        text: rawText,
        context: await getContextLexemes(rawText), // 检索词库中已有的相似条目
        task: "glossing_and_consistency_check"
    });

    // 4. 置信度决策逻辑
    const CONFIDENCE_THRESHOLD = userConfig.autoConfirmThreshold; // 用户设定的自动确认阈值，如 0.9
    
    const updatePayload = {
        transcription: aiResult.suggestedTranscription,
        ai_confidence: aiResult.confidenceScore,
        ai_reasoning: aiResult.reasoning,
        ai_mode: aiResult.confidenceScore >= CONFIDENCE_THRESHOLD ? 'AUTO' : 'SUGGEST'
    };

    // 5. 如果置信度极高，则全自动更新词库关联
    if (updatePayload.ai_mode === 'AUTO') {
        await db.utterances.update(utteranceId, { 
            ...updatePayload, 
            is_verified: true 
        });
        await syncToLexicon(aiResult.newLexemes); // 自动联动词库
    } else {
        // 否则，将其标记为“待审核”，并在 UI 上以橙色/红色高亮
        await db.utterances.update(utteranceId, { 
            ...updatePayload, 
            is_verified: false 
        });
    }

    return updatePayload;
}

2. 五大标签页的同步逻辑 (Sync Interactivity)
当你在不同标签页切换时，AI 助手需要保持“上下文感知”：
	•	[从转写页 -> 标注页]：选中 Utterance-A，切换到标注页时，应自动定位到该句子的形态拆分卡片。
	•	[从标注页 -> 词库页]：点击标注中的语素 kuš，词库页应立即跳转到对应的 Lexeme 编辑详情。
	•	[从分析页 -> 写作页]：在分析页通过 AI 验证的结论（如“否定词前置”），应能在写作页通过 AI 指令一键引用。

3. AI 审计与分析的触发逻辑 (Event Triggers)
为了实现您要求的“AI 辅助审计”，需要设计以下监听器：
	•	静默审计 (Passive Audit)：
	•	触发：用户在“标注页”修改了一个 Gloss。
	•	动作：后台 AI 扫描该词在其他 Utterances 中的标注。
	•	反馈：如果发现冲突，在“分析页”的审计列表增加一条记录，并在当前单元格显示红色波浪线。
	•	主动假设检验 (Active Analysis)：
	•	触发：用户在“分析页”输入指令。
	•	动作：AI 检索 corpus.utterances 数组，执行正则匹配或语义向量检索。
	•	反馈：更新 aiAssistant.messages 渲染可视化图表。

4. 离线一致性保证 (Sync Conflict)
由于您是独立开发，处理冲突的最佳方式是 “乐观更新 (Optimistic UI)”：
	•	逻辑：用户修改数据 -> 立即更新本地 Store 和 UI -> 后台异步写入 SQLite/RxDB。
	•	AI 介入：如果本地修改导致了违反语言学逻辑（如：给动词打了名词标签），AI 审计器在下次闲置时弹出“建议修正”。
