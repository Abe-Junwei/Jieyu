# 迁移蓝图：去双写 → `layer_segment_contents` 单一真源
Migration Blueprint: Eliminate Double-Write → Single Source of Truth in `layer_segment_contents`

> 文档状态：草案 | Date: 2026-03-27 | DB 当前版本：v27

---

## 目标 Goal

将翻译文本（translation text）的唯一权威存储从 `utterance_texts` 迁移至
`layer_segment_contents`，消除 BridgeService 的双写模式，最终废弃
`utterance_texts` 表。

- 默认转写层时间结构（utterance 边界）**不变**，仍存于 `utterances`
- 仅文本内容（translations/glosses/etc.）的存储路径被统一
- 独立层（`independent_boundary`、`time_subdivision`）本已直写 V2，**不受影响**

### 代码修改

**`src/db/index.ts`** — 新增 v28 升级脚本：

```typescript
// v28: Backfill utterance_texts → layer_segment_contents
// 将历史 utterance_texts 补全为 V2 条目，为去双写迁移铺底
this.version(28).stores({}).upgrade(async (tx: unknown) => {
  const typedTx = tx as any;
  const utteranceTextsTable = typedTx.table('utterance_texts');
  const utterancesTable     = typedTx.table('utterances');
  const layerSegmentsTable  = typedTx.table('layer_segments');
  const layerSegmentContentsTable = typedTx.table('layer_segment_contents');

  const [allTexts, allUtterances] = await Promise.all([
    utteranceTextsTable.toArray() as Promise<UtteranceTextDocType[]>,
    utterancesTable.toArray()     as Promise<UtteranceDocType[]>,
  ]);

  const utteranceById = new Map(allUtterances.map((u) => [u.id, u]));
  const now = new Date().toISOString();

  for (const text of allTexts) {
    const utt = utteranceById.get(text.utteranceId);
    if (!utt) continue;                                   // 孤立 text，跳过（Phase 3 数据清理）
    const segmentId = `segv2_${text.layerId}_${utt.id}`;
    const existing  = await layerSegmentsTable.get(segmentId);
    if (existing) continue;                               // 已有 V2 条目，无需补全

    await layerSegmentsTable.put({
      id:          segmentId,
      textId:      utt.textId,
      mediaId:     utt.mediaId || '__unknown_media__',
      layerId:     text.layerId,
      utteranceId: utt.id,
      startTime:   utt.startTime,
      endTime:     utt.endTime,
      provenance:  { actorType: 'system', method: 'projection', createdAt: now, updatedAt: now },
      createdAt:   now,
      updatedAt:   now,
    });
    await layerSegmentContentsTable.put({
      id:         text.id,
      textId:     utt.textId,
      segmentId,
      layerId:    text.layerId,
      modality:   text.modality,
      ...(text.text !== undefined     ? { text: text.text } : {}),
      ...(text.translationAudioMediaId ? { translationAudioMediaId: text.translationAudioMediaId } : {}),
      sourceType: text.sourceType,
      ...(text.ai_metadata ? { ai_metadata: text.ai_metadata } : {}),
      ...(text.provenance  ? { provenance:  text.provenance  } : {}),
      ...(text.accessRights ? { accessRights: text.accessRights } : {}),
      createdAt:  text.createdAt,
      updatedAt:  text.updatedAt,
    });
  }
});
```

// 按 content ID 批量查询 V2 翻译条目 | Batch fetch V2 translations by content id
async function getUtteranceTextsByIdsV2(
  db: JieyuDatabase,
  ids: string[],
): Promise<UtteranceTextDocType[]> {
  if (ids.length === 0) return [];
  const contents = await db.dexie.layer_segment_contents.bulkGet(ids);
  const validContents = contents.filter((c): c is LayerSegmentContentDocType => Boolean(c));
  if (validContents.length === 0) return [];
  const segmentIds = [...new Set(validContents.map((c) => c.segmentId))];
  const segments   = await db.dexie.layer_segments.bulkGet(segmentIds);
  const segmentById = new Map(
    segments.filter((s): s is LayerSegmentDocType => Boolean(s)).map((s) => [s.id, s]),
  );
  return validContents.map((c) => {
    const seg = segmentById.get(c.segmentId);
    return toLegacyLikeUtteranceText(seg!, c, seg?.utteranceId ?? '');
  }).filter((r) => r.utteranceId);
}
