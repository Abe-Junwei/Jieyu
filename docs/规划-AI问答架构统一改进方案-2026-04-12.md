### 1.2 五大核心瓶颈

| # | 瓶颈 | 现状数据 | 影响面 |
|---|------|---------|--------|
| **B1** | 上下文尾截断 | `maxContextChars: 2400`，`buildPromptContextBlock()` 超限直接砍尾 | LLM 丢失关键信息（当前选区、声学数据等），回答质量不稳定 |

## 三、统一改进方案

### 3.0 设计原则

1. **按需而非预注入**：本地上下文通过 Tool Call 按需获取，不预填满 prompt
2. **编辑优先**：AI 状态更新永远让位于用户交互
3. **渐进增强**：P0 → P4 逐阶段上线，每阶段独立可发布
4. **代价可见**：所有新增开销（token 消耗、延迟、信息损失）对用户透明可控
5. **强制校验**：本地自动工具白名单注册制 + schema 校验，不依赖约定

```ts
// Worker 内部：接收状态切片，按阈值重算 fingerprint
// 同时承担 embedding 搜索（现阻塞 10-50ms）与摘要 diff（P2 所需）

// ── 阈值策略：三选一命中即触发重算 | Threshold: any-of-three triggers recompute ──
const THRESHOLDS = {
  charDelta: 50,       // 累计字符变化 ≥ 50
  opCount: 5,          // 离散编辑操作 ≥ 5 次
  idleAfterEdit: 2000, // 最后一次编辑后静默 2s（用户停笔时算一次）
} as const;

let pendingDelta = { chars: 0, ops: 0 };
let idleTimer: ReturnType<typeof setTimeout> | null = null;
let lastFingerprint = '';

function computeFingerprint(state: StateSlice): string {
  // 轻量字段拼接（原 P0-C 方案内化至 Worker）
  return [
    state.aiChatEnabled ? '1' : '0',
    state.messagesLength,
    state.isStreaming ? '1' : '0',
    state.lastError ?? '',
    state.connectionTestStatus,
    state.pendingToolCallId ?? '',
    state.taskSessionStatus,
    state.totalSent,
    state.providerKind,
    state.model,
    state.acousticPhase ?? '',
    state.acousticBatchCount,
    state.calibrationState,
    state.providerAvailable ? '1' : '0',
  ].join('|');
}

function tryRecompute(state: StateSlice) {
  const fp = computeFingerprint(state);
  if (fp !== lastFingerprint) {
    lastFingerprint = fp;
    self.postMessage({ type: 'fingerprint', value: fp });
  }
}

self.onmessage = (e: MessageEvent) => {
  switch (e.data.type) {
    case 'state_slice': {
      const slice = e.data.payload as StateSlice;
      pendingDelta.chars += slice.charDelta ?? 0;
      pendingDelta.ops += 1;

      // 阈值检测 | Check thresholds
      if (pendingDelta.chars >= THRESHOLDS.charDelta ||
          pendingDelta.ops >= THRESHOLDS.opCount) {
        tryRecompute(slice);
        pendingDelta = { chars: 0, ops: 0 };
      }

      // idle 检测 | Reset idle timer
      if (idleTimer) clearTimeout(idleTimer);
      idleTimer = setTimeout(() => {
        tryRecompute(slice);
        pendingDelta = { chars: 0, ops: 0 };
      }, THRESHOLDS.idleAfterEdit);
      break;
    }

    case 'flush': {
      // 面板打开 → 立即用缓存的 delta 重算 | Panel opened → immediate recompute
      if (idleTimer) clearTimeout(idleTimer);
      tryRecompute(e.data.payload as StateSlice);
      pendingDelta = { chars: 0, ops: 0 };
      break;
    }

    case 'embedding_search': {
      const results = searchEmbeddingIndex(e.data.payload);
      self.postMessage({ type: 'embedding_result', results });
      break;
    }

    case 'summary_diff': {
      const similarity = computeCosineSimilarity(e.data.payload);
      self.postMessage({ type: 'summary_diff_result', similarity });
      break;
    }
  }
};
```

```ts
// localContextTools.ts
const LOCAL_TOOL_REGISTRY = new Map(LOCAL_CONTEXT_TOOLS.map(t => [t.name, t]));

export function executeLocalToolCall(
  name: string,
  args: unknown,
  ctx: AiPromptContext,
  callCountRef: { current: number },
): { result: unknown; error?: string } {
  // 1. 白名单校验
  const tool = LOCAL_TOOL_REGISTRY.get(name);
  if (!tool) return { result: null, error: `Unknown local tool: ${name}` };

  // 2. 速率限制（每轮 Agent Loop 内最多 20 次本地调用）
  if (callCountRef.current >= 20) return { result: null, error: 'Local tool call limit exceeded' };
  callCountRef.current++;

  // 3. 参数 schema 校验
  if (tool.params && !validateArgs(args, tool.params)) {
    return { result: null, error: `Invalid arguments for ${name}` };
  }

  return { result: tool.execute(ctx, args) };
}
```

export interface AgentLoopConfig {
  maxSteps: number;           // 上限（默认 6，参考 Vercel stopWhen）
  needsApproval: (tool) =>    // 高风险工具要求确认（参考 Vercel needsApproval）
    tool.meta?.destructive === true;
  prepareStep: (step) => {};  // 步间检查（参考 Vercel prepareStep）
}

export async function runAgentLoop(
  orchestrator: ChatOrchestrator,
  initialMessages: ChatMessage[],
  config: AgentLoopConfig,
): AsyncGenerator<AgentLoopEvent> {
  let step = 0;
  let messages = [...initialMessages];

  while (step < config.maxSteps) {
    const response = await orchestrator.stream(messages);
    yield { type: 'llm_response', response };

    if (!response.toolCall) break;           // 纯文本 → 结束
    if (config.needsApproval(response.toolCall)) {
      yield { type: 'needs_approval', toolCall: response.toolCall };
      const approved = await waitForApproval();
      if (!approved) break;
    }

    const result = await executeToolCall(response.toolCall);
    messages.push(
      { role: 'assistant', content: JSON.stringify(response.toolCall) },
      { role: 'user', content: JSON.stringify(result) },
    );
    yield { type: 'tool_result', result };
    step++;
  }

  if (step >= config.maxSteps) {
    yield { type: 'max_steps_reached' };
  }
}
```

```

const PROVIDER_CONTEXT_LIMITS: Record<string, number> = {
  'deepseek-chat':    64_000,

  'gemini-2.0-flash': 1_000_000,

export function computeContextBudget(
  providerKind: string,
  model: string,
): { systemBudget: number; historyBudget: number; toolResultBudget: number } {
  const totalTokens = PROVIDER_CONTEXT_LIMITS[model] ?? 16_000;
  const usableTokens = Math.min(totalTokens * 0.7, 64_000); // 预留 30% 给输出

    systemBudget: Math.min(2000, usableTokens * 0.15),
    historyBudget: Math.min(usableTokens * 0.5, 32_000),
    toolResultBudget: Math.min(usableTokens * 0.35, 24_000),
  };
}

### P4：离线 AI Provider（2-3 天，可选）

```ts
import { CreateMLCEngine } from '@mlc-ai/web-llm';
import type { LLMProvider, ChatMessage } from './LLMProvider';

export class WebLLMProvider implements LLMProvider {
  private engine: MLCEngine | null = null;
  private modelId = 'Qwen2-1.5B-Instruct-q4f16_1-MLC';  // ~800MB，适合离线

  async initialize(): Promise<void> {
    this.engine = await CreateMLCEngine(this.modelId, {
      initProgressCallback: (progress) => {
        // UI 显示模型下载/加载进度
      },
    });
  }

  async *streamChat(messages: ChatMessage[]): AsyncGenerator<string> {
    const chunks = await this.engine!.chat.completions.create({
      messages,
      stream: true,
      temperature: 0.7,
    });
    for await (const chunk of chunks) {
      const delta = chunk.choices[0]?.delta?.content;
      if (delta) yield delta;
    }
  }
}
```

**改动文件**：`src/ai/providers/providerCatalog.ts`
- 新增 `webllm` provider kind

- 适用场景：简单问答、单步工具调用、术语查询

**预期收益**：

## 六、风险与缓解

| 风险 | 等级 | 所在阶段 | 缓解措施 |

## 七、诚实边界

经过全面分析，原有 5 项"不能解决的问题"中 3 项已有明确技术路径（标记为"已缓解"），仅剩 **2 个硬边界**：

### 7.1 不可突破的硬边界

1. **LLM 幻觉不可消除**：无论本地/云端、无论上下文多大，生成式模型固有概率性错误无法彻底消除。缓解手段（Self-RAG、CRAG、tool grounding）只能降低而非归零
2. **音频直接理解缺失**：当前所有 Provider（含离线 WebLLM）不接受原始音频输入。音频理解必须先经 STT 转文本，再输入 LLM。等待多模态音频模型浏览器端可用（预计 2027+）

### 7.2 原有边界 → 已缓解

| 原有边界 | 缓解方案 | 所在阶段 |
|---------|---------|---------|
| 浏览器主线程天花板 | P0-A Worker 温热缓存，fingerprint + embedding 搜索 + 摘要 diff 全部移出主线程 | P0-A |
| 离线 LLM 缺失 | P4 WebLLM 离线 Provider，支持 Qwen2.5-1.5B 等小模型 | P4 |
| LLM 有限窗口 | P3 动态上下文窗口 + P2 摘要链，按 Provider 实际能力自适应 | P2+P3 |
| 单人维护带宽 | JSON 配置驱动（P3）减少硬编码；结构化 trace（P1-F）加速排障 | P1+P3 |
